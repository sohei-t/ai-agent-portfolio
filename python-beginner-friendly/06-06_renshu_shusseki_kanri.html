Step 4: 完成版プログラム

全てをまとめた完成版です。

# === 出欠管理プログラム ===
members = ["田中", "鈴木", "佐藤", "高橋", "渡辺"]
attendance = {
    "田中": True, "鈴木": True,
    "佐藤": False, "高橋": True,
    "渡辺": False
}

print(f"会議メンバー: {len(members)}人")
print("=== 出欠一覧 ===")
present_count = 0
absent_list = []
for name, present in attendance.items():
    if present:
        print(f"  {name}: 出席")
        present_count = present_count + 1
    else:
        print(f"  {name}: 欠席")
        absent_list.append(name)
print("================")
print(f"出席: {present_count}人")
print(f"欠席: {len(absent_list)}人")
if len(absent_list) > 0:
    print(f"欠席者: {absent_list}")
